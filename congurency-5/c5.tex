\item In the diagram below, $DE \perp AC, BF\perp AC, AB=CD, AE=CF$, prove that $\triangle{DEG}\cong\triangle{BFG}$.

\begin{asy}
import markers;
import math;
import geometry;
size(5cm);
pair A,B,C,D,EE,F,G,H;
G=(0,0);
A=(-4,0);
EE=(-2,0);
F=(2,0);
C=(4,0);
B=(2,3);
D=(-2,-3);
draw(B--D);
draw(D--EE);
draw(B--F);
draw(B--A--C--D);
label("$A$",A,W);
label("$B$",B,N);
label("$C$",C,E);
label("$D$",D,S);
label("$E$",EE,SW);
label("$F$",F,NE);
label("$G$",G,NW);
markrightangle(G,EE,D,size=2mm);
markrightangle(B,F,A,size=2mm);
\end{asy}

\item In the diagram below, $\angle{ADC}=90^\circ, BE\perp{AC}, AE=AD$. Prove that $\triangle{CEF}\cong\triangle{BDF}$.

\begin{asy}
import markers;
import math;
import geometry;
size(5cm);

C=(0,4);
F=(0,1.6);
EE=extension(B,F,A,C);
draw(A--D--C--A);
draw(B--EE);
draw(A--F);
draw(D--B);
label("$A$",A,W);
label("$B$",B,E);
label("$C$",C,E);
label("$D$",D,S);
label("$E$",EE,W);
label("$F$",F,NE);
markrightangle(A,EE,B,size=2mm);
markrightangle(C,D,A,size=2mm);
\end{asy}
